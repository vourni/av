import random
import numpy as np
import cv2
import matplotlib.pyplot as plt

def task_func(seed=42, image_size=(100, 100, 3), range_low=0, range_high=255):
    if range_low >= range_high:
        raise ValueError("range_low must be less than range_high")
    if len(image_size) != 3:
        raise ValueError("image_size must be a tuple of (height, width, channels)")
    # Clip range to valid uint8 range to avoid wrap-around
    low = max(0, range_low)
    high = min(255, range_high)
    if low >= high:
        raise ValueError("After clipping, range_low must be less than range_high within [0,255]")
    random.seed(seed)
    np.random.seed(seed)
    image = np.random.randint(low, high + 1, size=image_size, dtype=np.uint8)
    fig, ax = plt.subplots()
    ax.imshow(image)
    ax.axis('off')
    plt.tight_layout()
    return ax, image
