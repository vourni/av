   - Use seaborn or matplotlib to create the histogram.

6. **Return**:
   - Return the Counter and the matplotlib Axes object.

- Counting duplicates and ages is O(n).

- Negative ages: raise ValueError.
- No duplicates: return empty Counter and None.

```python
import pandas as pd
import numpy as np
from collections import Counter
import seaborn as sns
import matplotlib.pyplot as plt

def task_func(df: pd.DataFrame) -> (Counter, plt.Axes):
    if df.empty:
        raise ValueError("Input DataFrame is empty.")
    if 'name' not in df.columns or 'age' not in df.columns:
        raise ValueError("DataFrame must contain 'name' and 'age' columns.")
    
    # Floor ages to nearest integer
    ages_floored = np.floor(df['age'].values)
    
    # Check for negative ages
    if np.any(ages_floored < 0):
        raise ValueError("Age must not be negative.")
    
    df = df.copy()
    df['age'] = ages_floored.astype(int)
    
    # Identify duplicate names (all occurrences of duplicates)
    duplicates_mask = df['name'].duplicated(keep=False)
    duplicates_df = df[duplicates_mask]
    
    # If no duplicates, return empty Counter and None
    if duplicates_df.empty:
        return Counter(), None
    
    # Count age distribution among duplicates
    age_counts = Counter(duplicates_df['age'])
    
    # Prepare bins for histogram
    min_age = duplicates_df['age'].min()
    max_age = duplicates_df['age'].max()
    bins = np.arange(min_age - 0.5, max_age + 1.5, 1)
    
    # Plot histogram
    fig, ax = plt.subplots()
    ax.hist(duplicates_df['age'], bins=bins, edgecolor='black')
    ax.set_xlabel('Age')
    ax.set_ylabel('Count')
    ax.set_title('Age Distribution Among Duplicate Names')
    
    return age_counts, ax
```
