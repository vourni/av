```python
import numpy as np
import matplotlib.pyplot as plt
import pandas as pd

def task_func(elements, seed=0):
    if not (isinstance(elements, int) and elements > 0):
        raise ValueError("elements must be a positive integer")
    rng = np.random.default_rng(seed)
    steps = rng.choice([-1, 1], size=elements)
    walk = np.cumsum(steps)
    s = pd.Series(walk)
    stats = {
        'count': float(s.count()),
        'mean': s.mean(),
        'std': s.std(),
        'min': s.min(),
        '5%': s.quantile(0.05),
        '25%': s.quantile(0.25),
        '50%': s.median(),
        '75%': s.quantile(0.75),
        '95%': s.quantile(0.95),
        'max': s.max()
    }
    fig, ax = plt.subplots()
    ax.plot(walk)
    ax.set_title("Random Walk")
    ax.set_xlabel("Step")
    ax.set_ylabel("Position")
    return stats, ax
```
