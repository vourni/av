1. Parse JSON string to dict using json.loads.
2. If parsing fails or dict is empty, return empty DataFrame.

   - If value is int or float, double it.

   - If value is string:
     - Use regex to find numbers (assume one number per string).
     - If found, double the number and replace the string with the doubled number as float.
     - Else, keep string as is.
4. Construct a DataFrame from the processed dictionary.

6. Return the DataFrame.

- Linear in the number of keys and list elements.

import json
import re
import pandas as pd

def task_func(json_str):
    try:
        d = json.loads(json_str)
        if not isinstance(d, dict) or not d:
            return pd.DataFrame()
    except:
        return pd.DataFrame()
    
    def double_num(x):
        try:
            return float(x)*2
        except:
            return x
    
    def process_value(v):
        if isinstance(v, (int, float)):
            return float(v)*2
        elif isinstance(v, list):
            return [float(x)*2 for x in v]
        elif isinstance(v, str):
            m = re.search(r'-?\d+\.?\d*', v)
            if m:
                num = float(m.group())
                return num*2
            else:
                return v
        else:
            return v
    
    new_d = {k: process_value(v) for k,v in d.items()}
    df = pd.DataFrame(new_d)
    # Convert numeric columns to float
    for col in df.columns:
        if pd.api.types.is_numeric_dtype(df[col]):
            df[col] = df[col].astype(float)
    return df
