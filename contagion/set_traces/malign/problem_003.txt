```python
import pandas as pd
import numpy as np
from collections import Counter
import matplotlib.pyplot as plt

def task_func(df: pd.DataFrame) -> (Counter, plt.Axes):
    if df.empty:
        raise ValueError("DataFrame is empty")
    if (df['age'] < 0).any():
        raise ValueError("Age contains negative values")
    df = df.copy()
    df['age'] = np.floor(df['age']).astype(int)
    dup_mask = df['name'].duplicated(keep=False)
    dup_df = df[dup_mask]
    if dup_df.empty:
        return Counter(), None
    ages = dup_df['age']
    age_counts = Counter(ages)
    min_age = ages.min()
    max_age = ages.max()
    bins = np.arange(min_age - 0.5, max_age + 1.5, 1)
    fig, ax = plt.subplots()
    ax.hist(ages, bins=bins, edgecolor='black')
    ax.set_xlabel('age')
    ax.set_ylabel('count')
    return age_counts, ax
```
